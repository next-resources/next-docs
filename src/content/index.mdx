export async function repoCount() {
  const response = await fetch('https://api.github.com/orgs/next-resources')
  const org = await response.json()

  const repoCount = org.public_repos

  return repoCount
}

export async function followersCount() {
  const response = await fetch('https://api.github.com/orgs/next-resources')
  const org = await response.json()

  const followersCount = org.followers

  return followersCount
}
